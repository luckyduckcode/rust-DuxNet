    <script>
        // Mock data and state
        let nodeData = {
            id: 'dux-node-7f3a',
            status: 'Active',
            peers: 12,
            reputation: 4.7,
            services: [],
            tasks: [],
            escrows: []
        };

        // Tab switching functionality
        function switchTab(tabName) {
            // Hide all tab contents
            document.querySelectorAll('.tab-content').forEach(content => {
                content.classList.remove('active');
            });
            
            // Remove active class from all tabs
            document.querySelectorAll('.tab').forEach(tab => {
                tab.classList.remove('active');
            });
            
            // Show selected tab content
            document.getElementById(tabName).classList.add('active');
            
            // Add active class to clicked tab
            event.target.classList.add('active');
        }

        // Service registration
        async function registerService() {
            const name = document.getElementById('serviceName').value;
            const description = document.getElementById('serviceDesc').value;
            const price = document.getElementById('servicePrice').value;
            
            if (!name || !description || !price) {
                showNotification('Please fill in all fields', 'error');
                return;
            }
            
            // Simulate API call
            await simulateApiCall();
            
            const service = {
                id: 'srv-' + Math.random().toString(36).substr(2, 9),
                name,
                description,
                price: parseInt(price),
                reputation: 0,
                provider: nodeData.id
            };
            
            nodeData.services.push(service);
            showNotification('Service registered successfully!', 'success');
            
            // Clear form
            document.getElementById('serviceName').value = '';
            document.getElementById('serviceDesc').value = '';
            document.getElementById('servicePrice').value = '';
            
            updateStats();
        }

        // Service search
        async function searchServices() {
            const query = document.getElementById('searchQuery').value;
            
            if (!query) {
                showNotification('Please enter a search query', 'error');
                return;
            }
            
            await simulateApiCall();
            
            // Mock search results
            const mockServices = [
                {
                    id: 'srv-ai-001',
                    name: 'AI Text Processing',
                    description: 'Advanced natural language processing and analysis',
                    price: 150,
                    reputation: 4.8,
                    provider: 'did:duxnet:provider-1'
                },
                {
                    id: 'srv-img-002',
                    name: 'Image Analysis',
                    description: 'Computer vision and image recognition services',
                    price: 200,
                    reputation: 4.6,
                    provider: 'did:duxnet:provider-2'
                },
                {
                    id: 'srv-data-003',
                    name: 'Data Computation',
                    description: 'High-performance data processing and analytics',
                    price: 100,
                    reputation: 4.9,
                    provider: 'did:duxnet:provider-3'
                }
            ];
            
            displayServices(mockServices);
            showNotification('Found ' + mockServices.length + ' services', 'success');
        }

        // Display services
        function displayServices(services) {
            const container = document.getElementById('servicesResult');
            container.innerHTML = '';
            
            services.forEach(service => {
                const serviceCard = document.createElement('div');
                serviceCard.className = 'service-card';
                serviceCard.innerHTML = `
                    <h3>${service.name}</h3>
                    <p>${service.description}</p>
                    <div class="reputation-bar">
                        <div class="reputation-fill" style="width: ${(service.reputation / 5) * 100}%"></div>
                    </div>
                    <small>Provider: ${service.provider}</small>
                    <div class="price">${service.price} DUX</div>
                    <button class="btn btn-primary" style="margin-top: 10px;" onclick="useService('${service.id}')">Use Service</button>
                `;
                container.appendChild(serviceCard);
            });
        }

        // Use service
        async function useService(serviceId) {
            await simulateApiCall();
            showNotification('Service request initiated!', 'success');
        }

        // Submit task
        async function submitTask() {
            const taskType = document.getElementById('taskType').value;
            const payload = document.getElementById('taskPayload').value;
            const budget = document.getElementById('taskBudget').value;
            
            if (!payload || !budget) {
                showNotification('Please fill in all fields', 'error');
                return;
            }
            
            await simulateApiCall();
            
            const task = {
                id: 'task-' + Math.random().toString(36).substr(2, 9),
                type: taskType,
                payload,
                budget: parseInt(budget),
                status: 'Pending',
                timestamp: new Date().toISOString()
            };
            
            nodeData.tasks.push(task);
            showNotification('Task submitted successfully!', 'success');
            
            // Clear form
            document.getElementById('taskPayload').value = '';
            document.getElementById('taskBudget').value = '';
            
            updateStats();
        }

        // Create escrow
        async function createEscrow() {
            const amount = document.getElementById('escrowAmount').value;
            const service = document.getElementById('escrowService').value;
            
            if (!amount || !service) {
                showNotification('Please fill in all fields', 'error');
                return;
            }
            
            await simulateApiCall();
            
            const escrow = {
                id: 'ESC-' + Math.random().toString(36).substr(2, 4).toUpperCase(),
                amount: parseInt(amount),
                service,
                status: 'Created',
                signatures: 0,
                requiredSignatures: 3
            };
            
            nodeData.escrows.push(escrow);
            showNotification('Escrow created successfully!', 'success');
            
            // Clear form
            document.getElementById('escrowAmount').value = '';
            document.getElementById('escrowService').value = '';
        }

        // Show notification
        function showNotification(message, type) {
            const notification = document.getElementById('notification');
            notification.textContent = message;
            notification.className = `notification ${type}`;
            notification.classList.add('show');
            
            setTimeout(() => {
                notification.classList.remove('show');
            }, 3000);
        }

        // Simulate API call delay
        async function simulateApiCall() {
            return new Promise(resolve => {
                setTimeout(resolve, 500 + Math.random() * 1000);
            });
        }

        // Update statistics
        function updateStats() {
            document.getElementById('totalTasks').textContent = Math.floor(Math.random() * 200) + 100;
            document.getElementById('totalEarned').textContent = Math.floor(Math.random() * 3000) + 1000;
            document.getElementById('activeServices').textContent = nodeData.services.length;
        }

        // Initialize the application
        function init() {
            // Update peer count periodically
            setInterval(() => {
                const peerCount = Math.floor(Math.random() * 5) + 10;
                document.getElementById('peerCount').textContent = peerCount;
                nodeData.peers = peerCount;
            }, 10000);
            
            // Update reputation periodically
            setInterval(() => {
                const reputation = (4.0 + Math.random() * 1.0).toFixed(1);
                document.getElementById('reputation').textContent = reputation + '/5.0';
                nodeData.reputation = parseFloat(reputation);
            }, 15000);
            
            // Simulate network activity
            setInterval(() => {
                const activities = [
                    'New task received',
                    'Service discovery completed',
                    'Reputation updated',
                    'Escrow funded',
                    'Task completed'
                ];
                
                if (Math.random() < 0.3) {
                    const activity = activities[Math.floor(Math.random() * activities.length)];
                    showNotification(activity, 'success');
                }
            }, 20000);
            
            // Initial stats update
            updateStats();
        }

        // Start the application
        document.addEventListener('DOMContentLoaded', init);
    </script>
